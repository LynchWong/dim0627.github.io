tmuxでウィンドウを開いたり画面分割したときにディレクトリを維持する

こんな感じ。

# v でペインを縦に分割する
bind v split-window -h -c "#{pane_current_path}"

# s でペインを横に分割する
bind s split-window -v -c "#{pane_current_path}"

# 新規ウィンドウでディレクトリを保持
bind c new-window -c "#{pane_current_path}"
